// Pines de sensores
#define SENSOR_IZQ 2
#define SENSOR_DER 3

// Pines del Driver L298N
#define ENA 5   // PWM Motor Izquierdo
#define IN1 6
#define IN2 7
#define IN3 8
#define IN4 9
#define ENB 10  // PWM Motor Derecho

int velocidad = 150;  // Velocidad (0-255)

void setup() {
  pinMode(SENSOR_IZQ, INPUT);
  pinMode(SENSOR_DER, INPUT);
  pinMode(ENA, OUTPUT);
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
  pinMode(ENB, OUTPUT);
  Serial.begin(9600);

void loop() {
  int sensorIzq = digitalRead(SENSOR_IZQ);
  int sensorDer = digitalRead(SENSOR_DER);
  
  if (sensorIzq == LOW && sensorDer == LOW) {
    avanzar();
  }
  else if (sensorIzq == HIGH && sensorDer == LOW) {
    girarDerecha();
  }
  else if (sensorIzq == LOW && sensorDer == HIGH) {
    girarIzquierda();
  }
  else {
    detener();
  }

void avanzar() {
  analogWrite(ENA, velocidad);
  analogWrite(ENB, velocidad);
  digitalWrite(IN1, HIGH);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);

  analogWrite(ENA, velocidad);
  analogWrite(ENB, velocidad/2);

  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);

void girarIzquierda() {
  analogWrite(ENA, velocidad/2);
  analogWrite(ENB, velocidad);
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH);
  digitalWrite(IN4, LOW);
}

void detener() {
  analogWrite(ENA, 0);
  analogWrite(ENB, 0);
  digitalWrite(IN1, LOW);
  digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);
  digitalWrite(IN4, LOW);

#define VALVULA 13

char comando;

void setup() {
  Serial.begin(9600);
  
  pinMode(ENA, OUTPUT);
  pinMode(IN1, OUTPUT);
  pinMode(IN2, OUTPUT);
  pinMode(ENB, OUTPUT);
  pinMode(IN3, OUTPUT);
  pinMode(IN4, OUTPUT);
  pinMode(BOMBA, OUTPUT);
  pinMode(VALVULA, OUTPUT);
  
  servo1.attach(SERVO1);
  servo2.attach(SERVO2);
  servo3.attach(SERVO3);
  
  digitalWrite(BOMBA, LOW);
  digitalWrite(VALVULA, LOW);
}

void loop() {
  if (Serial.available() > 0) {
    comando = Serial.read();
    
    switch (comando) {
      case 'F':  // Adelante
        digitalWrite(IN1, HIGH);
        digitalWrite(IN2, LOW);
        digitalWrite(IN3, HIGH);
        digitalWrite(IN4, LOW);
        break;
      
      case 'B':  // Atr√°s
        digitalWrite(IN1, LOW);
        digitalWrite(IN2, HIGH);
        digitalWrite(IN3, LOW);
        digitalWrite(IN4, HIGH);
        break;
      
      case 'S':  // Stop
        digitalWrite(IN1, LOW);
        digitalWrite(IN2, LOW);
        digitalWrite(IN3, LOW);
        digitalWrite(IN4, LOW);
        break;
      
      case 'P':  // Proceso de siembra
        servo2.write(90); delay(500);  // Abrir surco
        servo1.write(45); delay(500);  // Soltar semilla
        servo3.write(90); delay(500);  // Tapar semilla
        servo1.write(0);
        servo2.write(0);
        servo3.write(0);
        break;
      
      case 'W':  // Activar riego
        digitalWrite(BOMBA, HIGH);
        digitalWrite(VALVULA, HIGH);
        delay(3000);  // Riega 3 segundos
        digitalWrite(BOMBA, LOW);
        digitalWrite(VALVULA, LOW);
        break;
    }
  }
}
